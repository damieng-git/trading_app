//@version=6
indicator("Future Swing [BigBeluga]", overlay = true, max_labels_count = 500, max_boxes_count = 500, max_bars_back = 5000)


// INPUTS ---------------------------------------------------------------------

len = input.int(
     30,
     "Swing Detection Length",
     minval = 10,
     tooltip = "Defines how many bars are used to detect swing highs and lows. Higher values produce fewer, stronger swings; lower values react faster but may create noise."
)

offset = input.int(
     10,
     "Future Projection Offset",
     tooltip = "How many bars into the future the projected swing target and zones are extended. This does not affect calculations, only visual projection."
)

dataqty = input.int(
     5,
     "Historical Swing Samples",
     minval = 3,
     maxval = 20,
     tooltip = "Number of completed historical swing percentage moves used to estimate the future swing projection. More samples smooth results but reduce responsiveness."
)

calcType = input.string(
     "Average",
     "Swing Projection Method",
     options = ["Average", "Median", "Mode"],
     tooltip = "Defines how historical swing percentages are aggregated:\n\n• Average – smooth, balanced projection\n• Median – ignores extreme outliers\n• Mode – focuses on the most frequent swing size"
)


string postion_d = input.string(
     "Bottom Right",
     title = "Dashboard Position",
     options = [
         "Middle Right",
         "Top Center",
         "Top Right",
         "Middle Left",
         "Middle Center",
         "Bottom Left",
         "Bottom Center",
         "Bottom Right"
     ],
     tooltip = "Sets the on-chart position of the statistics dashboard. Choose a location that avoids overlapping with price action and indicators."
)

string sizeData = input.string(
     "Normal",
     title = "Dashboard Text Size",
     options = ["Tiny", "Small", "Normal", "Large", "Huge"],
     tooltip = "Controls the text size of values displayed on the chart. Increase size for better readability on high-resolution screens."
)

DataSize = switch sizeData
    "Small" => size.small
    "Normal"=> size.normal
    "Large" => size.large
    "Huge"  => size.huge
    => size.tiny

dashPosition = switch postion_d
    "Middle Right" => position.middle_right
    "Middle Left" => position.middle_left
    "Middle Center" => position.middle_center

    "Top Right" => position.top_right
    "Top Left" => position.top_left
    "Top Center" => position.top_center

    "Bottom Right" => position.bottom_right
    "Bottom Left" => position.bottom_left
    "Bottom Center" => position.bottom_center


extendR = input.bool(
     false,
     "Extend Swing Zones Forward",
     tooltip = "If enabled, swing high and low zones continue extending to the right. When disabled, zones stop at the projection offset."
)

colorBull = input.color(color.rgb(21, 221, 124), "", inline = "1")
colorBear = input.color(color.rgb(235, 117, 20), "", inline = "1")

// CALCULATIONS ---------------------------------------------------------------
var swingPC = array.new<float>()

var swingUpper = box(na)
var swingLower = box(na)

// Structs for swing points
type swingD 
    float val 
    int indx 

var hSwing = swingD.new(na , na)
var lSwing = swingD.new(na , na)

var futureSwing = line(na)
var futureSwinglbl = label(na)


// Direction flag: true = down move, false = up move
var direc = false



// Detect swing highs and lows
H = ta.highest(len)
L = ta.lowest(len)

if high == H
    direc := true
if low == L
    direc := false

// Store completed swing high
if high[1] == H[1] and high < H 
    hSwing.indx := bar_index[1]
    hSwing.val  := high[1]

// Store completed swing low
if low[1] == L[1] and low > L 
    lSwing.indx := bar_index[1]
    lSwing.val  := low[1]

atr = ta.atr(200) * 0.5

// SWING LEG -------------------------------------------------------
buyVol = array.new<float>()
sellVol = array.new<float>()

if barstate.islast
    lowerIndex = math.min(hSwing.indx, lSwing.indx)


    for i = 0 to bar_index - lowerIndex

        c = close[i]
        o = open[i]
        v = volume[i]

        if c > o 
            buyVol.push(v)
        else 
            sellVol.push(v)

if direc != direc[1]

    mColor = direc ? colorBear : colorBull

    line.delete(line.new(hSwing.indx, hSwing.val, lSwing.indx, lSwing.val, color = mColor, width = 2)[1])


    swingUpper.delete()
    swingUpper := box.new(hSwing.indx, hSwing.val, bar_index + offset, hSwing.val+atr
                         , border_color = colorBear
                         , border_width = 1
                         , bgcolor = color.new(colorBear, 80)
                         , text_halign = text.align_right
                         , text_color = color.new(chart.fg_color, 10)
                         , text_size = DataSize)

    swingLower.delete()
    swingLower := box.new(lSwing.indx, lSwing.val, bar_index + offset, lSwing.val-atr
                         , border_color = colorBull
                         , border_width = 1
                         , bgcolor = color.new(colorBull, 80)
                         , text_halign = text.align_right
                         , text_color = color.new(chart.fg_color, 10)
                         , text_size = DataSize)


    pc = 
     not direc 
     ? (hSwing.val - lSwing.val) / lSwing.val * 100 
     :(lSwing.val - hSwing.val) / hSwing.val * 100 

    swingPC.push(pc)


swingUpper.set_right(bar_index+offset)
swingLower.set_right(bar_index+offset)

swingUpper.set_text(
    str.tostring(hSwing.val) + "🔺\n" +
    "-"+ str.tostring(sellVol.sum(), format.volume) + "🔻"
)

swingLower.set_text(
    str.tostring(lSwing.val) + "🔻\n" +
    "+"+ str.tostring(buyVol.sum(), format.volume) + "🔺"
)

if extendR
    swingLower.set_extend(extend.right)
    swingUpper.set_extend(extend.right)


if swingPC.size() > dataqty 
    swingPC.shift()



var dash = table.new(dashPosition, 2, 30, chart.bg_color)



// Real-time profile rebuilding while swing is still forming
if barstate.islast

    futureSwing.delete()
    futureSwinglbl.delete()
    mColor = chart.fg_color

    avgSwing = array.new<float>()


    if swingPC.size() > 0 

        for i = 0 to swingPC.size() - 1

            col = color.from_gradient(i, 0, swingPC.size() - 1, colorBear, colorBull)

            dash.cell(0, i, "Swing " + str.tostring(i+1), text_color = col, text_size = DataSize)
            dash.cell(1, i, str.tostring(swingPC.get(i), format.percent), text_color = col, text_size = DataSize)

            avgSwing.push(math.abs(swingPC.get(i)))

            if i == swingPC.size() - 1


                swingVal = switch calcType
                    "Average" =>avgSwing.avg()
                    "Median" =>avgSwing.median()
                    "Mode" =>avgSwing.mode()

                dash.cell(0, swingPC.size() - 1 + 1, "Swing AVG", text_color = chart.fg_color, text_size = DataSize)
                dash.cell(1, swingPC.size() - 1 + 1, (not direc ? "-" : "+") + str.tostring(swingVal, format.percent), text_color = chart.fg_color, text_size = DataSize)

                dash.cell(0, swingPC.size() - 1 + 2, "", text_color = chart.fg_color, text_size = DataSize)
                dash.cell(1, swingPC.size() - 1 + 2, "", text_color = chart.fg_color, text_size = DataSize)

                dash.cell(0, swingPC.size() - 1 + 3, "Delta Volume", text_color = chart.fg_color, text_size = DataSize)
                dash.cell(1, swingPC.size() - 1 + 3, str.tostring(buyVol.sum() - sellVol.sum(), format.volume), text_color = chart.fg_color, text_size = DataSize)

                dash.cell(0, swingPC.size() - 1 + 4, "Total Volume", text_color = chart.fg_color, text_size = DataSize)
                dash.cell(1, swingPC.size() - 1 + 4, str.tostring(buyVol.sum() + sellVol.sum(), format.volume), text_color = chart.fg_color, text_size = DataSize)



    swingVal = switch calcType
        "Average" =>avgSwing.avg()
        "Median" =>avgSwing.median()
        "Mode" =>avgSwing.mode()

    if not direc

        pChange = hSwing.val - ( hSwing.val * (swingVal/100))

        futureSwinglbl := label.new(
                        bar_index+offset, pChange
                        , "-" + str.tostring(swingVal, format.percent)
                        , style = label.style_label_left
                        , color = color.new(chart.fg_color, 80)
                        , textcolor = chart.fg_color, size = DataSize
                        )

        futureSwing := line.new(hSwing.indx, hSwing.val, bar_index+offset, pChange, color = mColor, style = line.style_dashed)
      
    // Mirror logic for bullish direction
    else

        pChange = lSwing.val + ( lSwing.val * (swingVal/100))

        futureSwinglbl := label.new(
                        bar_index+offset, pChange
                        , "+" + str.tostring(swingVal, format.percent)
                        , style = label.style_label_left
                        , color = color.new(chart.fg_color, 80)
                        , textcolor = chart.fg_color, size = DataSize
                        )
                    
        futureSwing := line.new(bar_index+offset, pChange, lSwing.indx, lSwing.val, color = mColor, style = line.style_dashed)
